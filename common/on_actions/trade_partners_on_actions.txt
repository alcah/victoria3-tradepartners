on_monthly_pulse_country = {
	on_actions = {
 		on_actions_trade_partners_global_loop
	}
}

on_game_started_after_lobby = {
	on_actions = {
		on_actions_trade_partners_game_init
	}
}

on_actions_trade_partners_global_loop = {
	effect = {
		every_market = {
			limit = {
				is_player = yes
			}
			trade_partners_execute_market_update = yes
		}
	}
}

	effect = {
		every_market = {
			limit = {
				is_player = yes
			}
			trade_partners_execute_market_update = yes
		}
	}
}
