on_monthly_pulse = {
	on_actions = {
 		on_actions_trade_partners_market_update
	}
}

on_game_started_after_lobby = {
	on_actions = {
		on_actions_trade_partners_market_update
	}
}

on_actions_trade_partners_market_update = {
	effect = {
		every_market = {
			limit = {
				market_trade = { value > 0 }
			}
			trade_partners_market_update = yes
		}
	}
}
