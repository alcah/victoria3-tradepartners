macro = {
	description = "Get imports from root to target market"
	definition = "TradePartnersGetImportsFrom(Importer, Exporter)"
	replace_with = "GuiScope.SetRoot(Importer.MakeScope).AddScope('base_market', Exporter.MakeScope).ScriptValue('trade_partners_total_imports_from_target')"
}

macro = {
	description = "Get exports from root to target market"
	definition = "TradePartnersGetExportsTo(Exporter, Importer)"
	replace_with = "GuiScope.SetRoot(Exporter.MakeScope).AddScope('base_market', Importer.MakeScope).ScriptValue('trade_partners_total_exports_to_target')"
}

macro = {
	description = "Check if the markets trade"
	definition = "TradePartnersImportsFromP(Importer, Exporter)"
	replace_with = "GreaterThanOrEqualTo_CFixedPoint(TradePartnersGetImportsFrom(Importer, Exporter), '(CFixedPoint)1')"
}

macro = {
	description = "Check if the markets trade"
	definition = "TradePartnersExportsToP(Exporter, Importer)"
	replace_with = "GreaterThanOrEqualTo_CFixedPoint(TradePartnersGetExportsTo(Importer, Exporter), '(CFixedPoint)1')"
}
